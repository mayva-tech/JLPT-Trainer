### File — `C:\Projects\jlpt-n2-video-player\src\pages\PlayerPage.tsx`

**Action:** Replace. **Purpose:** The 16:9 stage rendering all six step-cards in sequence. Manual prev/next advances whole *items* for now; Step 2 adds within-item step advancement + audio.

```tsx
import { useState } from "react";
import { getLessonById } from "../data/lessons";
import { getVocabularyByIds } from "../data/vocabulary";
import type { StepName } from "../types/player";
import { CategoryCard } from "../components/CategoryCard";
import { WordCard } from "../components/WordCard";
import { PhraseCard } from "../components/PhraseCard";
import { SentenceCard } from "../components/SentenceCard";
import { ShadowingCard } from "../components/ShadowingCard";
import { ReviewCard } from "../components/ReviewCard";
import { ProgressIndicator } from "../components/ProgressIndicator";

const STEPS: StepName[] = [
  "category",
  "word",
  "phrase",
  "sentence",
  "shadowing",
  "review",
];

export function PlayerPage() {
  const lesson = getLessonById("lesson-01");
  const items = lesson ? getVocabularyByIds(lesson.vocabularyIds) : [];

  const [itemIndex, setItemIndex] = useState(0);
  const [stepIndex, setStepIndex] = useState(0);

  if (!lesson || items.length === 0) {
    return (
      <div className="stage-wrapper">
        <div className="stage">
          <div className="safe-area">
            <div className="word-meaning">No vocabulary found.</div>
          </div>
        </div>
      </div>
    );
  }

  const item = items[itemIndex]!;
  const step = STEPS[stepIndex] as StepName;

  function goNextStep() {
    if (stepIndex < STEPS.length - 1) {
      setStepIndex((s) => s + 1);
    } else if (itemIndex < items.length - 1) {
      setItemIndex((i) => i + 1);
      setStepIndex(0);
    }
  }

  function goPrevStep() {
    if (stepIndex > 0) {
      setStepIndex((s) => s - 1);
    } else if (itemIndex > 0) {
      setItemIndex((i) => i - 1);
      setStepIndex(STEPS.length - 1);
    }
  }

  const isFirst = itemIndex === 0 && stepIndex === 0;
  const isLast =
    itemIndex === items.length - 1 && stepIndex === STEPS.length - 1;

  function renderStep() {
    switch (step) {
      case "category":
        return <CategoryCard item={item} />;
      case "word":
        return <WordCard item={item} />;
      case "phrase":
        return <PhraseCard item={item} />;
      case "sentence":
        return <SentenceCard item={item} />;
      case "shadowing":
        return <ShadowingCard item={item} phase="listen" />;
      case "review":
        return <ReviewCard item={item} />;
    }
  }

  return (
    <>
      <div className="stage-wrapper">
        <div className="stage">
          <ProgressIndicator
            current={itemIndex}
            total={items.length}
            step={step}
          />
          {renderStep()}
        </div>
      </div>

      <div className="step-indicator">
        Step {stepIndex + 1} of {STEPS.length} — {step}
      </div>

      <div className="nav-bar">
        <button onClick={goPrevStep} disabled={isFirst}>
          ← Back
        </button>
        <button onClick={goNextStep} disabled={isLast}>
          Forward →
        </button>
      </div>
    </>
  );
}
```
